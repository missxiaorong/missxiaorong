js/jq 两种方式分别实现返回顶部效果(2016.1.11)

上一篇文章：

现在大多数网站都有返回顶部的效果，下面的总结分别是js和Jq实现返回顶部按钮。本页面使用的是原生js实现的效果。

html代码：

<body> <div class="main"> <img src="images/123.png"> <img src="images/123.png"> </div> <div id="back-to-top"> <a style="display: none;" href="javascript:;" class="btn" id="btn" title="返回顶部"> </a> </div> </body>

css代码：（其中的随意的颜色设置是观看效果，你可以随意改变，例如本页面将背景颜色换成了你看到的图片）

<style> *{ margin: 0; padding: 0; } /*js示例中的样式*/ .btn{ width: 30px; height: 30px; position: fixed; bottom: 30px; left: 50%; margin-left: 500px; background-color: #0066ff; } .btn:hover{ background-color: red; } /*jQuery示例中的样式*/ #back-to-top{ position:fixed; display:none; bottom:100px; right:80px; width: 30px;; height: 30px;; background-color: red; } </style>

js代码

window.onload=function(){ //获取返回顶部id var topbtn=document.getElementById("btn"); var timer = null; //添加滚动条监控事件 window.onscroll=function(){ //获取页面可见区域宽度 var pagelookheight = document.documentElement.clientHeight; //获取页面被卷去的高度即（滚动条距离页面开始顶端高度） var backtop = document.body.scrollTop; if(backtop>pagelookheight){ topbtn.style.display ="block" //滚动超过第一屏时返回按钮显示 }else{ topbtn.style.display = "none" //滚动条在页面第一屏时返回按钮不存在 } } //返回按钮点击事件 topbtn.onclick=function(){ //alert(11); //设置计时器设定滚动时间 timer = setInterval(function(){ var backtop = document.body.scrollTop; var speedtop = backtop / 5; //变化速度 document.body.scrollTop = backtop-speedtop;//变化高度 if(backtop==0){ clearInterval(timer); } },30) } }

jquery代码：

<script src="http://ajax.microsoft.com/ajax/jQuery/jquery-1.7.2.min.js"></script> $(function(){ $(window).scroll(function(){ if ($(window).scrollTop()>400){ $("#back-to-top").fadeIn(1500); } else { $("#back-to-top").fadeOut(1500); } }); $("#back-to-top").click(function(){ $('body,html').animate({scrollTop:0},1000); return false; }); });

相比较这两个效果，个人觉得jquery的animate()动画效果对用户更友好，当然你可以随意选择自己喜欢的一种。
